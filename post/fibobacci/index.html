package main

import "fmt"

func main() {
	var number int
	for {
		if num, err := fmt.Scanf("%d", &number); num != 1 || err != nil {
			return
		}
		begin, next := 0, 1
		for begin+next < number {
			begin, next = next, begin+next
		}
		if number-next < begin+next-number {
			fmt.Println(number - next)
		} else {
			fmt.Println(begin + next - number)
		}
	}
}
